Como usar GROUP BY en coding-----------------------------------------------------------------------

SELECT 
    name_store,
    COUNT(name) AS name_cnt,
    COUNT(DISTINCT name) AS name_uniq_cnt
FROM
    products_data_all
GROUP BY
    name_store;	
----------------------------------------------------------------------------------------------------
Consulta para calcular el peso maximo de producto para cada categoria

SELECT 
    category,
    MAX(CAST(weight AS real)) AS max_weight
FROM
    products_data_all
GROUP BY
    category;
-----------------------------------------------------------------------------------------------------
Calcula el precio promedio del producto, precio minimo y precio maximo en cada tienda de la tabla

SELECT 
    name_store,
    AVG(price) AS average_price,
    MAX(price) AS max_price,
    MIN(price) AS min_price
FROM
    products_data_all
GROUP BY
    name_store;
------------------------------------------------------------------------------------------------------
calcula la diferencia entre los precios maximos y minimos de cada producto en la categoria `milk`

SELECT 
    name,
    MAX(price) - MIN(price) AS max_min_diff
FROM
    products_data_all
WHERE
    category = 'milk'
    AND date_upd :: date = '2019-06-10'
GROUP BY
    name;
